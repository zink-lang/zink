39/// Create a new zink project.
40#[derive(Debug, Parser)]
41pub struct New {
42    path: PathBuf,
43
44    /// The name of the project.
45    #[clap(short, long, value_name = "name")]
46    name: Option<String>,
47}
48
49impl New {
50    /// Get the project name.
51    pub fn name(&self) -> Result<String> {
52        let name = if let Some(name) = self.name.as_ref() {
53            name.into()
54        } else {
55            self.path
56                .file_name()
57                .and_then(|n| n.to_str())
58                .ok_or_else(|| anyhow!("Invalid project path"))?
59                .into()
60        };
61
62        Ok(name)
63    }
64
65    /// Create a new zink project.
66    pub fn run(&self) -> Result<()> {
67        fs::create_dir_all(&self.path)?;
68
69        // Create the cargo manifest.
70        let mut manifest = Manifest::default();
71        let name = self.name()?;
72        manifest.name(&name);
73        fs::write(
74            self.path.join("Cargo.toml"),
75            toml::to_string_pretty(&manifest)?,
76        )?;
77
78        // Create the src directory.
79        let src = self.path.join("src");
80        fs::create_dir_all(&src)?;
81        fs::write(
82            src.join("lib.rs"),
83            ADDITION.trim_start().replace(NAME, &name),
84        )?;
85
86        // Create README
87        fs::write(
88            self.path.join("README.md"),
89            README.trim_start().replace(NAME, &name),
90        )?;
91
92        println!("{} zink project `{}`", "Created".green().bold(), name);
93        Ok(())
94    }
95}
